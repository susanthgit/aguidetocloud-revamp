{{/* render-image.html — Markdown image render hook. When a markdown page writes `![alt](path.webp)`, Hugo routes through this template. We auto-emit a with AVIF source + WebP fallback for any .webp source. scripts/build-images.mjs guarantees the AVIF sibling exists in static/. Other formats (PNG, JPG, GIF, SVG) flow through unchanged. */}} {{- $dest := .Destination | safeURL -}} {{- $alt := .PlainText -}} {{- $title := .Title -}} {{- if hasSuffix (lower $dest) ".webp" -}} {{- $avifSrc := replaceRE `\.webp$` ".avif" $dest -}} {{- else -}} {{- end -}}
